> dfs 思路 - 四个方向一直遍历到底
```js

cconst rl = require("readline").createInterface({ input: process.stdin });
var iter = rl[Symbol.asyncIterator]();
const readline = async () => (await iter.next()).value;

let visited = []
let map = []
let count = 0 
let directions = [[0,1], [0,-1],[1,0], [-1,0]]
let rows = 0
let cols = 0 

function dfs(r, c){
    for(let d of directions){
        let nextR = r + d[0]
        let nextC = c + d[1]
        if(nextC < 0 || nextC >= cols || nextR <0 || nextR >= rows) continue 
        if(map[nextR][nextC] == 1 && !visited[nextR][nextC]){
                visited[nextR][nextC]  = true 
                dfs(nextR, nextC)
        }
    }
    
}


void async function () {
    let firstLine = (await readline()).split(' ').map(Number)
    map = new Array(firstLine[0]).fill(0)
    visited = new Array(firstLine[0]).fill(0).map(() => new Array(firstLine[1]).fill(false))
    let level = 0 
    while(line = await readline()){
        let cur = line.split(' ').map(Number)
        map[level] = [...cur]
        level++
    }
    
    rows = firstLine[0]
    cols = firstLine[1]
    
    for(let r = 0; r < rows; r++){
        for(let c = 0; c < cols; c++){
            if(map[r][c] == 1 && !visited[r][c]) {
                visited[r][c] = true 
                count ++ 
                dfs(r, c)
            }
        }
    }
    console.log(count)
}()
```

```js
const rl = require("readline").createInterface({ input: process.stdin });
var iter = rl[Symbol.asyncIterator]();
const readline = async () => (await iter.next()).value;

let visited = []
let map = []
let count = 0 
let directions = [[0,1], [0,-1],[1,0], [-1,0]]
let rows = 0
let cols = 0 

function bfs(r, c){
    let queue = []
    queue.push([r,c])
        
    while(queue.length != 0){
        let cur = queue.shift()
        for(let d of directions){
            let nextR = cur[0] + d[0]
            let nextC = cur[1] + d[1]
            if(nextC < 0 || nextC >= cols || nextR <0 || nextR >= rows) continue 
            if(map[nextR][nextC] == 1 && !visited[nextR][nextC]){
                    visited[nextR][nextC]  = true 
                    queue.push([nextR,nextC])
            }
        }
        
    }
    
}

    let firstLine = (await readline()).split(' ').map(Number)
    map = new Array(firstLine[0]).fill(0)
    visited = new Array(firstLine[0]).fill(0).map(() => new Array(firstLine[1]).fill(false))
    let level = 0 
    while(line = await readline()){
        let cur = line.split(' ').map(Number)
        map[level] = [...cur]
        level++
    }
    
    rows = firstLine[0]
    cols = firstLine[1]
    
    for(let r = 0; r < rows; r++){
        for(let c = 0; c < cols; c++){
            if(map[r][c] == 1 && !visited[r][c]) {
                visited[r][c] = true 
                count ++ 
                bfs(r, c)
            }
        }
    }
    console.log(count)
}()
```
