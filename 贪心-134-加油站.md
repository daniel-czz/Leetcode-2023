/** 暴力解法 
 * @param {number[]} gas
 * @param {number[]} cost
 * @return {number}
 */
var canCompleteCircuit = function(gas, cost) {
    for(let i = 0; i < cost.length; i++) {
        let rest = gas[i] - cost[i]  //记录剩余油量
        // 以i为起点行驶一圈，index为下一个目的地
        let index = i+1 == gas.length? 0: i+1
        while(rest > 0 && index !== i) {
            rest += gas[index] - cost[index]
            index = index+1 == gas.length? 0: index+1
        }
        if(rest >= 0 && index === i) return i
    }
    return -1
};

/**
 * @param {number[]} gas
 * @param {number[]} cost
 * @return {number}
 */
 // totalCost > totalGas 肯定没有解 
 // 要不然题目一定有解 整个arr走一遍就好了，
 // 只要找到一个能走到底（arr最后一位的）的就是解，就不需要模拟转一圈了 
var canCompleteCircuit = function(gas, cost) {
    let totalCost = 0
    let totalGas = 0
    let curSum = 0 
    let start = 0 
    for(let i = 0; i < gas.length; i++) {
        totalCost = totalCost + cost[i]
        totalGas = totalGas + gas[i]
        curSum = curSum + gas[i] - cost[i] 
        // 当 curSum < 0 时，说明从 start 到 i 的路已经失败。
	    // 由于 油箱从 start 开始累积，仍然无法到达 i+1，
        // 所以 0 ~ i 之间的任何位置都不可能是成功的起点。
	    // 唯一可能的起点是 i+1，因为 0 ~ i 的任何站点都会因为之前的消耗而失败。
        if(curSum < 0) { 
            start = i + 1  
            curSum = 0 
        }
    }
    if(totalCost > totalGas) return -1 
    return start 
};
